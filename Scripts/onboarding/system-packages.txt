# ── ydotool 1.x build deps (E18) ──
# Ubuntu 24.04 only ships ydotool 0.1.8 in apt, which lacks --absolute mousemove
# (needed to translate CU click coords correctly). install.sh builds v1.0.4 from
# source so we get the absolute API. Daemon writes to /dev/uinput at the kernel
# layer — works on both Wayland (where xdotool can't reach native apps) and X11.

libevdev-dev             # SHOULD_HAVE # ydotool 1.x build (input event lib headers)
